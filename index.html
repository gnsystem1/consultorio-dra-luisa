/** * GN Systems - backend completo com agendamentos, pacientes e bloqueios. * * ABAS: * AGENDAMENTOS * PACIENTES * BLOQUEIOS * * BLOQUEIOS: * ID | Data | Tipo | HoraInicial | HoraFinal | Motivo | Status | CriadoEm * * Tipos: * dia_todo * horario * periodo */ const ABA_AGENDAMENTOS = 'AGENDAMENTOS'; const ABA_PACIENTES = 'PACIENTES'; const ABA_BLOQUEIOS = 'BLOQUEIOS'; const HEADERS_AGENDAMENTOS = [ 'ID','Paciente','Data','Hora','E-mail','Telefone','Idade','Sexo','TipoConsulta','Profissional', 'Status','Confirmado','Observacoes','DataCadastro','Lembrete24h','DataLembrete24h','DataConclusao', 'QueixaPrincipal','ResumoAtendimento','Conduta','ObservacoesClinicas','RetornoSugerido' ]; const HEADERS_PACIENTES = [ 'ID','Paciente','E-mail','Telefone','Idade','Sexo','Observacoes','DataCadastro' ]; const HEADERS_BLOQUEIOS = [ 'ID','Data','Tipo','HoraInicial','HoraFinal','Motivo','Status','CriadoEm' ]; function doGet(e) { const acao = getParam_(e, 'acao'); const data = getParam_(e, 'data'); const ignorarId = getParam_(e, 'ignorarId'); if (acao === 'horarios_ocupados') { const info = getDisponibilidadeDoDia_(data, ignorarId); return jsonOutput_({ horarios: info.horarios, diaTodo: info.diaTodo }); } return jsonOutput_({ agendamentos: getSheetValues_(ABA_AGENDAMENTOS, HEADERS_AGENDAMENTOS), pacientes: getSheetValues_(ABA_PACIENTES, HEADERS_PACIENTES), bloqueios: getSheetValues_(ABA_BLOQUEIOS, HEADERS_BLOQUEIOS) }); } function doPost(e) { try { const payload = JSON.parse((e && e.postData && e.postData.contents) || '{}'); const acao = String(payload.acao || ''); if (acao === 'criar_agendamento') return criarAgendamento_(payload); if (acao === 'atualizar_agendamento') return atualizarAgendamento_(payload); if (acao === 'criar_bloqueio') return criarBloqueio_(payload); if (acao === 'remover_bloqueio') return removerBloqueio_(payload.id); return jsonOutput_({ status: 'erro', mensagem: 'Ação inválida.' }); } catch (err) { return jsonOutput_({ status: 'erro', mensagem: err.message || 'Falha inesperada.' }); } } function criarAgendamento_(payload) { const sheet = getSheet_(ABA_AGENDAMENTOS, HEADERS_AGENDAMENTOS); const data = normalizarData_(payload.data); const hora = normalizarHora_(payload.hora); if (!data || !hora || !String(payload.paciente || '').trim()) { return jsonOutput_({ status: 'erro', mensagem: 'Paciente, data e hora são obrigatórios.' }); } const disponibilidade = getDisponibilidadeDoDia_(data); if (disponibilidade.diaTodo) { return jsonOutput_({ status: 'erro', mensagem: 'Esta data está bloqueada o dia inteiro.' }); } if (disponibilidade.horarios.indexOf(hora) !== -1) { return jsonOutput_({ status: 'erro', mensagem: 'Esse horário já está ocupado ou bloqueado.' }); } const row = [ Utilities.getUuid(), String(payload.paciente || '').trim(), data, hora, String(payload.email || '').trim(), digitsOnly_(payload.telefone), String(payload.idade || ''), String(payload.sexo || ''), String(payload.tipo_consulta || ''), String(payload.profissional || ''), String(payload.status || 'Confirmado'), String(payload.confirmado || 'Sim'), String(payload.observacoes || ''), new Date(), 'Pendente', '', '', '', '', '', '', '' ]; sheet.appendRow(row); const pacienteRow = upsertPacienteFromPayload_(payload); return jsonOutput_({ status: 'ok', agendamento: row, paciente: pacienteRow }); } function atualizarAgendamento_(payload) { const sheet = getSheet_(ABA_AGENDAMENTOS, HEADERS_AGENDAMENTOS); const values = sheet.getDataRange().getValues(); for (var i = 1; i < values.length; i++) { if (String(values[i][0]) !== String(payload.id || '')) continue; const dataAtual = normalizarData_(payload.data || values[i][2]); const horaAtual = normalizarHora_(payload.hora || values[i][3]); const mudouDataHora = String(dataAtual) !== normalizarData_(values[i][2]) || String(horaAtual) !== normalizarHora_(values[i][3]); if (mudouDataHora) { const disponibilidade = getDisponibilidadeDoDia_(dataAtual, payload.id); if (disponibilidade.diaTodo) { return jsonOutput_({ status: 'erro', mensagem: 'Esta data está bloqueada o dia inteiro.' }); } if (disponibilidade.horarios.indexOf(horaAtual) !== -1) { return jsonOutput_({ status: 'erro', mensagem: 'Esse horário já está ocupado ou bloqueado.' }); } } values[i][2] = dataAtual; values[i][3] = horaAtual; if (payload.status !== undefined) values[i][10] = String(payload.status || values[i][10]); if (payload.confirmado !== undefined) values[i][11] = String(payload.confirmado || values[i][11]); if (payload.observacoes !== undefined) values[i][12] = String(payload.observacoes || ''); if (payload.queixa_principal !== undefined) values[i][17] = String(payload.queixa_principal || ''); if (payload.resumo_atendimento !== undefined) values[i][18] = String(payload.resumo_atendimento || ''); if (payload.conduta !== undefined) values[i][19] = String(payload.conduta || ''); if (payload.observacoes_clinicas !== undefined) values[i][20] = String(payload.observacoes_clinicas || ''); if (payload.retorno_sugerido !== undefined) values[i][21] = String(payload.retorno_sugerido || ''); if (normalizeText_(values[i][10]).indexOf('real') !== -1 && !values[i][16]) { values[i][16] = new Date(); } sheet.getRange(i + 1, 1, 1, HEADERS_AGENDAMENTOS.length).setValues([values[i]]); return jsonOutput_({ status: 'ok', agendamento: values[i] }); } return jsonOutput_({ status: 'erro', mensagem: 'Agendamento não encontrado.' }); } function criarBloqueio_(payload) { const sheet = getSheet_(ABA_BLOQUEIOS, HEADERS_BLOQUEIOS); const data = normalizarData_(payload.data); const tipo = String(payload.tipo || 'dia_todo').trim(); const horaInicial = normalizarHora_(payload.horaInicial); const horaFinal = normalizarHora_(payload.horaFinal); if (!data) { return jsonOutput_({ status: 'erro', mensagem: 'Informe a data do bloqueio.' }); } if (tipo !== 'dia_todo' && tipo !== 'horario' && tipo !== 'periodo') { return jsonOutput_({ status: 'erro', mensagem: 'Tipo de bloqueio inválido.' }); } if (tipo === 'horario' && !horaInicial) { return jsonOutput_({ status: 'erro', mensagem: 'Informe o horário do bloqueio.' }); } if (tipo === 'periodo') { if (!horaInicial || !horaFinal) { return jsonOutput_({ status: 'erro', mensagem: 'Informe hora inicial e hora final.' }); } if (timeToMinutes_(horaFinal) < timeToMinutes_(horaInicial)) { return jsonOutput_({ status: 'erro', mensagem: 'Hora final deve ser maior que a inicial.' }); } } const validacao = validarNovoBloqueio_(data, tipo, horaInicial, horaFinal); if (!validacao.ok) { return jsonOutput_({ status: 'erro', mensagem: validacao.mensagem }); } const row = [ Utilities.getUuid(), data, tipo, tipo === 'dia_todo' ? '' : horaInicial, tipo === 'periodo' ? horaFinal : (tipo === 'horario' ? horaInicial : ''), String(payload.motivo || '').trim(), 'Ativo', new Date() ]; sheet.appendRow(row); return jsonOutput_({ status: 'ok', bloqueio: row }); } function removerBloqueio_(id) { const sheet = getSheet_(ABA_BLOQUEIOS, HEADERS_BLOQUEIOS); const values = sheet.getDataRange().getValues(); for (var i = 1; i < values.length; i++) { if (String(values[i][0]) === String(id || '')) { values[i][6] = 'Inativo'; sheet.getRange(i + 1, 1, 1, HEADERS_BLOQUEIOS.length).setValues([values[i]]); return jsonOutput_({ status: 'ok', bloqueio: values[i] }); } } return jsonOutput_({ status: 'erro', mensagem: 'Bloqueio não encontrado.' }); } function validarNovoBloqueio_(data, tipo, horaInicial, horaFinal) { const bloqueios = getBloqueiosDoDia_(data); const agendamentos = getAgendamentosDoDia_(data); if (tipo === 'dia_todo') { const existeAtendimento = agendamentos.some(function(r) { return !isCancelado_(r[10]); }); if (existeAtendimento) { return { ok: false, mensagem: 'Já existem agendamentos ativos nessa data.' }; } const existeBloqueioAtivo = bloqueios.some(function(r) { return isBloqueioAtivoRow_(r); }); if (existeBloqueioAtivo) { return { ok: false, mensagem: 'Essa data já possui bloqueio cadastrado.' }; } return { ok: true }; } const disponibilidade = getDisponibilidadeDoDia_(data); if (disponibilidade.diaTodo) { return { ok: false, mensagem: 'Essa data já está bloqueada o dia inteiro.' }; } const novosSlots = tipo === 'periodo' ? generateSlots_(horaInicial, horaFinal) : [horaInicial]; const conflitoAgendamento = agendamentos.some(function(r) { if (isCancelado_(r[10])) return false; return novosSlots.indexOf(normalizarHora_(r[3])) !== -1; }); if (conflitoAgendamento) { return { ok: false, mensagem: 'Já existe agendamento em um dos horários informados.' }; } const slotsBloqueados = disponibilidade.horarios || []; const conflitoBloqueio = novosSlots.some(function(h) { return slotsBloqueados.indexOf(h) !== -1; }); if (conflitoBloqueio) { return { ok: false, mensagem: 'Já existe bloqueio em um dos horários informados.' }; } return { ok: true }; } function getDisponibilidadeDoDia_(data, ignorarId) { const dateValue = normalizarData_(data); const agendamentos = getAgendamentosDoDia_(dateValue, ignorarId); const bloqueios = getBloqueiosDoDia_(dateValue, ignorarId).filter(isBloqueioAtivoRow_); const diaTodo = bloqueios.some(function(r) { return normalizeText_(r[2]) === 'dia_todo'; }); const horarios = []; agendamentos.forEach(function(r) { if (!isCancelado_(r[10])) { const hora = normalizarHora_(r[3]); if (hora) horarios.push(hora); } }); if (!diaTodo) { bloqueios.forEach(function(r) { const tipo = normalizeText_(r[2]); const horaInicial = normalizarHora_(r[3]); const horaFinal = normalizarHora_(r[4]); if (tipo === 'horario' && horaInicial) { horarios.push(horaInicial); } else if (tipo === 'periodo' && horaInicial && horaFinal) { Array.prototype.push.apply(horarios, generateSlots_(horaInicial, horaFinal)); } }); } return { diaTodo: diaTodo, horarios: uniqueSorted_(horarios) }; } function getAgendamentosDoDia_(data, ignorarId) { const values = getSheetValues_(ABA_AGENDAMENTOS, HEADERS_AGENDAMENTOS); return values.slice(1).filter(function(r) { return normalizarData_(r[2]) === data && String(r[0]) !== String(ignorarId || ''); }); } function getBloqueiosDoDia_(data, ignorarId) { const values = getSheetValues_(ABA_BLOQUEIOS, HEADERS_BLOQUEIOS); return values.slice(1).filter(function(r) { return normalizarData_(r[1]) === data && String(r[0]) !== String(ignorarId || ''); }); } function upsertPacienteFromPayload_(payload) { const sheet = getSheet_(ABA_PACIENTES, HEADERS_PACIENTES); const values = sheet.getDataRange().getValues(); const nome = String(payload.paciente || '').trim(); const email = String(payload.email || '').trim(); const telefone = digitsOnly_(payload.telefone); for (var i = 1; i < values.length; i++) { const sameName = normalizeText_(values[i][1]) === normalizeText_(nome); const samePhone = digitsOnly_(values[i][3]) && digitsOnly_(values[i][3]) === telefone; const sameEmail = String(values[i][2] || '').trim() && String(values[i][2] || '').trim().toLowerCase() === email.toLowerCase(); if (sameName || samePhone || sameEmail) { values[i][1] = nome || values[i][1]; values[i][2] = email || values[i][2]; values[i][3] = telefone || values[i][3]; values[i][4] = String(payload.idade || values[i][4] || ''); values[i][5] = String(payload.sexo || values[i][5] || ''); if (payload.observacoes) values[i][6] = String(payload.observacoes); sheet.getRange(i + 1, 1, 1, HEADERS_PACIENTES.length).setValues([values[i]]); return values[i]; } } const row = [ Utilities.getUuid(), nome, email, telefone, String(payload.idade || ''), String(payload.sexo || ''), String(payload.observacoes || ''), new Date() ]; sheet.appendRow(row); return row; } function getSheetValues_(name, headers) { return getSheet_(name, headers).getDataRange().getValues(); } function getSheet_(name, headers) { const ss = SpreadsheetApp.getActive(); let sheet = ss.getSheetByName(name); if (!sheet) sheet = ss.insertSheet(name); if (sheet.getLastRow() === 0 && headers && headers.length) { sheet.appendRow(headers); } return sheet; } function jsonOutput_(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function getParam_(e, key) { return String((e && e.parameter && e.parameter[key]) || ''); } function normalizarData_(value) { if (value === null || value === undefined || value === '') return ''; if (Object.prototype.toString.call(value) === '[object Date]' && !isNaN(value)) { return Utilities.formatDate(value, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } const s = String(value).trim(); if (/^\d{4}-\d{2}-\d{2}$/.test(s)) return s; if (/^\d{4}-\d{2}-\d{2}T/.test(s)) return s.slice(0, 10); if (/^\d{1,2}\/\d{1,2}\/\d{4}$/.test(s)) { const parts = s.split('/'); return [ parts[2], ('0' + parts[1]).slice(-2), ('0' + parts[0]).slice(-2) ].join('-'); } const d = new Date(s); if (!isNaN(d)) { return Utilities.formatDate(d, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } return ''; } function normalizarHora_(value) { if (value === null || value === undefined || value === '') return ''; if (Object.prototype.toString.call(value) === '[object Date]' && !isNaN(value)) { return Utilities.formatDate(value, Session.getScriptTimeZone(), 'HH:mm'); } const s = String(value).trim(); const match = s.match(/(\d{1,2}):(\d{2})/); if (!match) return ''; return ('0' + match[1]).slice(-2) + ':' + match[2]; } function timeToMinutes_(hora) { const h = normalizarHora_(hora); if (!h) return -1; const parts = h.split(':'); return Number(parts[0]) * 60 + Number(parts[1]); } function generateSlots_(horaInicial, horaFinal) { const start = timeToMinutes_(horaInicial); const end = timeToMinutes_(horaFinal); if (start < 0 || end < 0 || end < start) return []; const out = []; for (var m = start; m <= end; m += 30) { const hh = ('0' + Math.floor(m / 60)).slice(-2); const mm = ('0' + (m % 60)).slice(-2); out.push(hh + ':' + mm); } return out; } function uniqueSorted_(list) { const map = {}; list.forEach(function(item) { if (item) map[item] = true; }); return Object.keys(map).sort(); } function digitsOnly_(value) { return String(value || '').replace(/\D/g, ''); } function normalizeText_(value) { return String(value || '') .toLowerCase() .normalize('NFD') .replace(/[\u0300-\u036f]/g, ''); } function isCancelado_(status) { return normalizeText_(status).indexOf('cancel') !== -1; } function isBloqueioAtivoRow_(row) { return normalizeText_(row[6]) !== 'inativo'; }
